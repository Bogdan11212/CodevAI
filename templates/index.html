Пример работы CodevAI
def fibonacci(n):
    """
    Функция вычисляет n-ое число Фибоначчи.
    
    Args:
        n: Порядковый номер числа Фибоначчи
        
    Returns:
        int: n-ое число Фибоначчи
    """
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n-1) + fibonacci(n-2)
